import PyTango

# replace with hostname of the device
host = "hostname"

# connect to device over GPIB/Ethernet
dev = PyTango.DeviceProxy("tango://{}:10000/sys/tg_test/1".format(host))

# clear device
dev.Clear()

# reset device
dev.Reset()

# get device identification
response = dev.IdnQuery()
print("Device found:", response)

# configure frequency measurement
dev.Write(":CONF:FREQ 1MHz, 0.001 HZ, (@1)")
